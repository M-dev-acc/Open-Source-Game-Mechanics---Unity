/// <summary>
/// Represents a single node in a dialogue tree, containing dialogue text, the speaker's name, and choices.
/// </summary>
[System.Serializable]

// Adds the ability to create a DialogueNode ScriptableObject from the Unity Editor's 'Create' menu.
// Useful for constructing modular and reusable dialogue structures.
[CreateAssetMenu(fileName = "NewDialogue", menuName = "Dialogue System/Dialogue")]
public class DialogueNode : ScriptableObject
{
    /// <summary>
    /// Name of the speaker for this dialogue node.
    /// </summary>
    public string speakerName;

    /// <summary>
    /// The dialogue text displayed to the player.
    /// Supports multi-line input in the inspector.
    /// </summary>
    [TextArea(2, 5)] public string dialogueText;

    /// <summary>
    /// List of choices available to the player at this node.
    /// </summary>
    public List<DialogueChoice> dialogueChoices;
}
